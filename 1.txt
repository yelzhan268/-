# Создание файла "Домашние_животные" с содержимым
echo -e "Собаки\nКошки\nХомяки" > Домашние_животные

# Создание файла "Вьючные_животные" с содержимым
echo -e "Лошади\nВерблюды\nОслы" > Вьючные_животные

# Объединение файлов
cat Домашние_животные Вьючные_животные > Друзья_человека

# Просмотр содержимого созданного файла
cat Друзья_человека

# Переименование файла
mv Друзья_человека Друзья_человека.txt
