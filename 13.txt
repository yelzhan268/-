class Животные:
    def __init__(self, имя, дата_рождения):
        self.имя = имя
        self.дата_рождения = дата_рождения

class ДомашниеЖивотные(Животные):
    def __init__(self, имя, команда, дата_рождения):
        super().__init__(имя, дата_рождения)
        self.команда = команда

class ВьючныеЖивотные(Животные):
    def __init__(self, имя, команда, дата_рождения):
        super().__init__(имя, дата_рождения)
        self.команда = команда
